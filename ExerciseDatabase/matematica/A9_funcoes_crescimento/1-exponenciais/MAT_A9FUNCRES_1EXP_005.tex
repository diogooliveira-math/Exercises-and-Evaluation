\exercicio{
Número de bactérias: $f(t)=400\times 3^{t/2}$ para $t$ horas após 7h.
\begin{enumerate}
  \item Calcula número às 8h e às 11h.
  \item Determina tempo até 2400 bactérias (horas e minutos).
  \item Esboça o gráfico e interpreta três pontos.
  \item Cria um problema adicional de crescimento e resolve.
\end{enumerate}
}
