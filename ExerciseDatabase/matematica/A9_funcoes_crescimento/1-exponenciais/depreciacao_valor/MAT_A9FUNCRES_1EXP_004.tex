\exercicio{

\begin{enumerate}
  \item Estima valor aos 18 meses.
  \item Determina quando $f(t)<4000$.
  \item Calcula percentagem de desvalorização anual.
  \item Esboça o gráfico e interpreta três pontos.
\end{enumerate}
