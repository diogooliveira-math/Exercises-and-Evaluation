\exercicio{
Considera a função $f(x) = x^2 - 4x + 3$.
}

\subexercicio{Calcula a derivada $f'(x)$.}

\subexercicio{Determina os zeros de $f'(x)$, ou seja, resolve $f'(x) = 0$.}

\subexercicio{Estuda o sinal de $f'(x)$ e indica os intervalos onde a função é crescente e decrescente.}

\subexercicio{Identifica o ponto onde ocorre um extremo (máximo ou mínimo) e classifica-o.}
