\begin{center}
\begin{tabular}{|l|c|}
\hline
\textbf{Fruta} & \textbf{Quantidade} \\
\hline
Maçãs & 45 \\
Bananas & 38 \\
Laranjas & 22 \\
Peras & 15 \\
\hline
\end{tabular}
\end{center}
}
